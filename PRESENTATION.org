* Addendum

** FFI is great:

   - Import values or types (values shown below)

#+BEGIN_SRC purescript
-- src/Math.purs
module Math where
foreign import pow :: Number -> Number -> Number
#+END_SRC

#+BEGIN_SRC js
// src/Math.js
"use strict";
exports.pow = function(x) {
  return function(y) {
    return Math.pow(x,y);
  };
};
#+END_SRC
